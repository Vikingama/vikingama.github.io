CRA 的 Webpack 配置

2020.04.27

明明写的最多的是业务代码，为什么配置 Webpack 会成为前端必备技能？

目录结构

.
├── config
│   ├── env.js                              运行环境配置
│   ├── getHttpsConfig.js                   获取 HTTPS 配置
│   ├── modules.js                          模块配置，例如路径别名、TS 配置...
│   ├── paths.js                            对路径进行处理，没太懂...
│   ├── pnpTs.js                            TS 文件的编译配置，没太懂...
│   ├── webpack.config.js                   Webpack 配置
│   └── webpackDevServer.config.js          DevServer 配置
├── package.json                            项目依赖
├── public
├── scripts
│   ├── build.js                            运行打包任务
│   └── start.js                            运行开发任务
└── src                                     业务代码

webpack.config.js

'use strict';

const fs = require('fs');
const path = require('path');
const webpack = require('webpack');
const resolve = require('resolve');
// ts-pnp 的 Webpack 插件
const PnpWebpackPlugin = require('pnp-webpack-plugin');
const HtmlWebpackPlugin = require('html-webpack-plugin');
// 防止不同的系统下对于路径大小写的问题导致的出错
const CaseSensitivePathsPlugin = require('case-sensitive-paths-webpack-plugin');
const InlineChunkHtmlPlugin = require('react-dev-utils/InlineChunkHtmlPlugin');
const TerserPlugin = require('terser-webpack-plugin'); // 压缩 JS
const MiniCssExtractPlugin = require('mini-css-extract-plugin'); // 将 CSS 单独提取出来
const OptimizeCSSAssetsPlugin = require('optimize-css-assets-webpack-plugin'); // 压缩 CSS
const safePostCssParser = require('postcss-safe-parser'); // 修复 CSS 语法错误
const ManifestPlugin = require('webpack-manifest-plugin'); // 生成 Manifest 文件，管理所有所需模块之间的交互
const InterpolateHtmlPlugin = require('react-dev-utils/InterpolateHtmlPlugin'); // 允许你在 index.html 中使用变量
const WorkboxWebpackPlugin = require('workbox-webpack-plugin'); // 将需要缓存的资源注入到 service worker 中
const WatchMissingNodeModulesPlugin = require('react-dev-utils/WatchMissingNodeModulesPlugin'); // 安装缺失的模块之后，自动重启 Webpack
const ModuleScopePlugin = require('react-dev-utils/ModuleScopePlugin'); // 禁止引入项目目录之外的模块
const getCSSModuleLocalIdent = require('react-dev-utils/getCSSModuleLocalIdent');
const paths = require('./paths');
const modules = require('./modules');
const getClientEnvironment = require('./env');
const ModuleNotFoundPlugin = require('react-dev-utils/ModuleNotFoundPlugin');
const ForkTsCheckerWebpackPlugin = require('react-dev-utils/ForkTsCheckerWebpackPlugin');
const typescriptFormatter = require('react-dev-utils/typescriptFormatter');
const postcssNormalize = require('postcss-normalize'); // 根据浏览器配置项，引入部分样式重置文件
const appPackageJson = require(paths.appPackageJson);
const shouldUseSourceMap = process.env.GENERATE_SOURCEMAP !== 'false'; // 是否生成 SourceMap
const shouldInlineRuntimeChunk = process.env.INLINE_RUNTIME_CHUNK !== 'false';
const isExtendingEslintConfig = process.env.EXTEND_ESLINT === 'true';
const imageInlineSizeLimit = parseInt(
  process.env.IMAGE_INLINE_SIZE_LIMIT || '10000' // 将图片打包的阈值
);
const useTypeScript = fs.existsSync(paths.appTsConfig);
// 匹配样式文件的正则
const cssRegex = /\.css$/;
const cssModuleRegex = /\.module\.css$/;
const sassRegex = /\.(scss|sass)$/;
const sassModuleRegex = /\.module\.(scss|sass)$/;
// 开发和打包的公用配置
module.exports = function (webpackEnv) {
  const isEnvDevelopment = webpackEnv === 'development';
  const isEnvProduction = webpackEnv === 'production';
  const isEnvProductionProfile =
    isEnvProduction && process.argv.includes('--profile');

  // 提供对 %PUBLIC_URL% 和 process.env.PUBLIC_URL 的支持
  const env = getClientEnvironment(paths.publicUrlOrPath.slice(0, -1));

  // 配置样式 Loader
  const getStyleLoaders = (cssOptions, preProcessor) => {
    const loaders = [
      // 如果在开发环境，将 CSS 注入 HTML
      isEnvDevelopment && require.resolve('style-loader'),
      // 生产环境压缩 CSS
      isEnvProduction && {
        loader: MiniCssExtractPlugin.loader,
        options: paths.publicUrlOrPath.startsWith('.')
          ? { publicPath: '../../' }
          : {}
      },
      {
        loader: require.resolve('css-loader'),
        options: cssOptions
      },
      {
        loader: require.resolve('postcss-loader'),
        options: {
          ident: 'postcss',
          plugins: () => [
            require('postcss-flexbugs-fixes'),
            require('postcss-preset-env')({
              autoprefixer: {
                flexbox: 'no-2009'
              },
              stage: 3
            }),
            postcssNormalize()
          ],
          sourceMap: isEnvProduction && shouldUseSourceMap
        }
      }
    ].filter(Boolean);
    if (preProcessor) {
      loaders.push(
        {
          loader: require.resolve('resolve-url-loader'),
          options: {
            sourceMap: isEnvProduction && shouldUseSourceMap
          }
        },
        {
          loader: require.resolve(preProcessor),
          options: {
            sourceMap: true
          }
        }
      );
    }
    return loaders;
  };

  return {
    // Webpack 运行模式
    mode: isEnvProduction ? 'production' : isEnvDevelopment && 'development',
    // 在出现错误时停止编译抛出错误，默认情况下会提示错误继续打包
    bail: isEnvProduction,
    // 对开发者工具的支持
    devtool: isEnvProduction
      ? shouldUseSourceMap
        ? 'source-map'
        : false
      : isEnvDevelopment && 'cheap-module-source-map',
    // 项目入口
    entry: [
      isEnvDevelopment &&
        require.resolve('react-dev-utils/webpackHotDevClient'),
      paths.appIndexJs
    ].filter(Boolean),
    output: {
      // 生产环境下，将项目打包到 paths.appBuild 目录
      path: isEnvProduction ? paths.appBuild : undefined,
      // 在 bundle 中引入「所包含模块信息」的相关注释
      pathinfo: isEnvDevelopment,
      // 打包生成的文件名，生产环境默认根据文件名和文件内容哈希值设置
      filename: isEnvProduction
        ? 'static/js/[name].[contenthash:8].js'
        : isEnvDevelopment && 'static/js/bundle.js',
      futureEmitAssets: true,
      // 代码分割时，分割出去的代码块命名规则
      chunkFilename: isEnvProduction
        ? 'static/js/[name].[contenthash:8].chunk.js'
        : isEnvDevelopment && 'static/js/[name].chunk.js',
      // 对于按需加载或加载外部资源来说是很重要的选项
      publicPath: paths.publicUrlOrPath,
      // 自定义每个 source map 的 sources 数组中使用的名称
      devtoolModuleFilenameTemplate: isEnvProduction
        ? (info) =>
            path
              .relative(paths.appSrc, info.absoluteResourcePath)
              .replace(/\\/g, '/')
        : isEnvDevelopment &&
          ((info) =>
            path.resolve(info.absoluteResourcePath).replace(/\\/g, '/')),
      // 用于按需加载 chunk 的 JSONP 函数，如果在同一网页中使用了多个（来自不同编译过程的）Webpack Runtime，则需要修改此选项
      jsonpFunction: `webpackJsonp${appPackageJson.name}`,
      // 设置模块中的顶级对象，在浏览器中指向 window，在 Node 中指向 global
      globalObject: 'this'
    },
    // 自定义优化，只在生产环境生效
    optimization: {
      minimize: isEnvProduction, // 使用 TerserPlugin 压缩 bundle
      minimizer: [
        // 通过提供一个或多个定制过的 TerserPlugin 实例，覆盖默认压缩工具
        new TerserPlugin({
          terserOptions: {
            parse: {
              ecma: 8
            },
            compress: {
              ecma: 5,
              warnings: false,
              comparisons: false,
              inline: 2
            },
            mangle: {
              safari10: true
            },
            keep_classnames: isEnvProductionProfile,
            keep_fnames: isEnvProductionProfile,
            output: {
              ecma: 5,
              comments: false,
              ascii_only: true
            }
          },
          sourceMap: shouldUseSourceMap
        }),
        new OptimizeCSSAssetsPlugin({
          cssProcessorOptions: {
            parser: safePostCssParser,
            map: shouldUseSourceMap
              ? {
                  inline: false,
                  annotation: true
                }
              : false
          },
          cssProcessorPluginOptions: {
            preset: ['default', { minifyFontValues: { removeQuotes: false } }]
          }
        })
      ],
      // 对于动态导入模块，默认使用全新的通用分块策略
      splitChunks: {
        chunks: 'all', // 将动态导入的所有模块分块
        name: false
      },
      /*
        runtime 是 Webpack 的运行环境（具体作用就是模块解析、加载）和模块信息清单，模块信息清单在每次有模块变更时都会变更，
        把这部分代码单独打包出来，配合后端缓存策略，这样就不会因为某个模块的变更导致包含模块信息的模块缓存失效，
        optimization.runtimeChunk 就是告诉 Webpack 是否要把这部分单独打包出来
      */
      runtimeChunk: {
        name: (entrypoint) => `runtime-${entrypoint.name}`
      }
    },
    // 模块解析细节
    resolve: {
      // 解析模块时应该搜索的目录
      modules: ['node_modules', paths.appNodeModules].concat(
        modules.additionalModulePaths || []
      ),
      // 自动解析确定的扩展
      extensions: paths.moduleFileExtensions
        .map((ext) => `.${ext}`)
        .filter((ext) => useTypeScript || !ext.includes('ts')),
      // 创建 import 或 require 的别名，来确保模块引入变得更简单
      alias: {
        'react-native': 'react-native-web',
        ...(isEnvProductionProfile && {
          'react-dom$': 'react-dom/profiling',
          'scheduler/tracing': 'scheduler/tracing-profiling'
        }),
        ...(modules.webpackAliases || {})
      },
      // 额外的解析插件列表
      plugins: [
        PnpWebpackPlugin,
        new ModuleScopePlugin(paths.appSrc, [paths.appPackageJson])
      ]
    },
    resolveLoader: {
      plugins: [PnpWebpackPlugin.moduleLoader(module)]
    },
    // 对不同模块的处理
    module: {
      strictExportPresence: true, // 文件中如果缺少导出时会直接报错而不是警告
      rules: [
        {
          // 禁用 require.ensure
          parser: {
            requireEnsure: false
          }
        },
        {
          test: /\.(js|mjs|jsx|ts|tsx)$/,
          // 首先进行 ESLint 检查
          enforce: 'pre',
          use: [
            {
              options: {
                cache: true,
                formatter: require.resolve(
                  // 默认的 ESLint 配置，在对警告/错误的信息做了美化
                  'react-dev-utils/eslintFormatter'
                ),
                eslintPath: require.resolve('eslint'),
                resolvePluginsRelativeTo: __dirname
              },
              loader: require.resolve('eslint-loader')
            }
          ],
          include: paths.appSrc
        },
        {
          // 当规则匹配时，只匹配下列中的一个，都不满足则应用最后一条规则
          oneOf: [
            {
              test: [/\.bmp$/, /\.gif$/, /\.jpe?g$/, /\.png$/],
              loader: require.resolve('url-loader'),
              options: {
                limit: imageInlineSizeLimit,
                name: 'static/media/[name].[hash:8].[ext]'
              }
            },
            {
              test: /\.(js|mjs|jsx|ts|tsx)$/,
              include: paths.appSrc, // 只处理 src 目录下的文件
              loader: require.resolve('babel-loader'),
              options: {
                customize: require.resolve(
                  'babel-preset-react-app/webpack-overrides'
                ),
                plugins: [
                  [
                    require.resolve('babel-plugin-named-asset-import'),
                    {
                      loaderMap: {
                        svg: {
                          ReactComponent:
                            '@svgr/webpack?-svgo,+titleProp,+ref![path]'
                        }
                      }
                    }
                  ]
                ],
                // 启用 babel-loader 缓存
                cacheDirectory: true,
                cacheCompression: false,
                compact: isEnvProduction
              }
            },
            {
              test: /\.(js|mjs)$/,
              exclude: /@babel(?:\/|\\{1,2})runtime/,
              loader: require.resolve('babel-loader'),
              options: {
                babelrc: false,
                configFile: false,
                compact: false,
                presets: [
                  [
                    require.resolve('babel-preset-react-app/dependencies'),
                    { helpers: true }
                  ]
                ],
                cacheDirectory: true,
                cacheCompression: false,
                sourceMaps: shouldUseSourceMap,
                inputSourceMap: shouldUseSourceMap
              }
            },
            {
              test: cssRegex,
              exclude: cssModuleRegex,
              use: getStyleLoaders({
                importLoaders: 1,
                sourceMap: isEnvProduction && shouldUseSourceMap
              }),
              sideEffects: true
            },
            // 默认支持 CSS Module
            {
              test: cssModuleRegex,
              use: getStyleLoaders({
                importLoaders: 1,
                sourceMap: isEnvProduction && shouldUseSourceMap,
                modules: {
                  getLocalIdent: getCSSModuleLocalIdent
                }
              })
            },
            // 默认支持 SASS
            {
              test: sassRegex,
              exclude: sassModuleRegex,
              use: getStyleLoaders(
                {
                  importLoaders: 3,
                  sourceMap: isEnvProduction && shouldUseSourceMap
                },
                'sass-loader'
              ),
              sideEffects: true
            },
            // 同时支持 CSS Module 和 SASS
            {
              test: sassModuleRegex,
              use: getStyleLoaders(
                {
                  importLoaders: 3,
                  sourceMap: isEnvProduction && shouldUseSourceMap,
                  modules: {
                    getLocalIdent: getCSSModuleLocalIdent
                  }
                },
                'sass-loader'
              )
            },
            {
              loader: require.resolve('file-loader'),
              exclude: [/\.(js|mjs|jsx|ts|tsx)$/, /\.html$/, /\.json$/],
              options: {
                name: 'static/media/[name].[hash:8].[ext]'
              }
            }
          ]
        }
      ]
    },
    plugins: [
      // 生成 HTML 文件
      new HtmlWebpackPlugin(
        Object.assign(
          {},
          {
            inject: true,
            //  模板目录
            template: paths.appHtml
          },
          isEnvProduction
            ? {
                minify: {
                  removeComments: true,
                  collapseWhitespace: true,
                  removeRedundantAttributes: true,
                  useShortDoctype: true,
                  removeEmptyAttributes: true,
                  removeStyleLinkTypeAttributes: true,
                  keepClosingSlash: true,
                  minifyJS: true,
                  minifyCSS: true,
                  minifyURLs: true
                }
              }
            : undefined
        )
      ),
      // 将 runtime 脚本嵌入到 HTML 中，这个文件很小，不会阻塞网络请求
      isEnvProduction &&
        shouldInlineRuntimeChunk &&
        new InlineChunkHtmlPlugin(HtmlWebpackPlugin, [/runtime-.+[.]js/]),
      // 注入一些环境变量到 index.html
      new InterpolateHtmlPlugin(HtmlWebpackPlugin, env.raw),
      // 当模块丢失时，给予必要的提示
      new ModuleNotFoundPlugin(paths.appPath),
      // 使一些环境变量在 JS 代码中可访问 if (process.env.NODE_ENV === 'production') { ... }...
      new webpack.DefinePlugin(env.stringified),
      // HMR 的支持
      isEnvDevelopment && new webpack.HotModuleReplacementPlugin(),
      isEnvDevelopment && new CaseSensitivePathsPlugin(),
      // npm install 之后不需要重启 Webpack
      isEnvDevelopment &&
        new WatchMissingNodeModulesPlugin(paths.appNodeModules),
      isEnvProduction &&
        new MiniCssExtractPlugin({
          filename: 'static/css/[name].[contenthash:8].css',
          chunkFilename: 'static/css/[name].[contenthash:8].chunk.css'
        }),
      new ManifestPlugin({
        fileName: 'asset-manifest.json',
        publicPath: paths.publicUrlOrPath,
        generate: (seed, files, entrypoints) => {
          const manifestFiles = files.reduce((manifest, file) => {
            manifest[file.name] = file.path;
            return manifest;
          }, seed);
          const entrypointFiles = entrypoints.main.filter(
            (fileName) => !fileName.endsWith('.map')
          );

          return {
            files: manifestFiles,
            entrypoints: entrypointFiles
          };
        }
      }),
      // 对于 moment.js 的优化
      new webpack.IgnorePlugin(/^\.\/locale$/, /moment$/),
      // 生成 service worker 脚本
      isEnvProduction &&
        new WorkboxWebpackPlugin.GenerateSW({
          clientsClaim: true,
          exclude: [/\.map$/, /asset-manifest\.json$/],
          importWorkboxFrom: 'cdn',
          navigateFallback: paths.publicUrlOrPath + 'index.html',
          navigateFallbackBlacklist: [
            new RegExp('^/_'),
            new RegExp('/[^/?]+\\.[^/]+$')
          ]
        }),
      // TypeScript 类型检查
      useTypeScript &&
        new ForkTsCheckerWebpackPlugin({
          typescript: resolve.sync('typescript', {
            basedir: paths.appNodeModules
          }),
          async: isEnvDevelopment,
          useTypescriptIncrementalApi: true,
          checkSyntacticErrors: true,
          resolveModuleNameModule: process.versions.pnp
            ? `${__dirname}/pnpTs.js`
            : undefined,
          resolveTypeReferenceDirectiveModule: process.versions.pnp
            ? `${__dirname}/pnpTs.js`
            : undefined,
          tsconfig: paths.appTsConfig,
          reportFiles: [
            '**',
            '!**/__tests__/**',
            '!**/?(*.)(spec|test).*',
            '!**/src/setupProxy.*',
            '!**/src/setupTests.*'
          ],
          silent: true,
          formatter: isEnvProduction ? typescriptFormatter : undefined
        })
    ].filter(Boolean),
    node: {
      module: 'empty',
      dgram: 'empty',
      dns: 'mock',
      fs: 'empty',
      http2: 'empty',
      net: 'empty',
      tls: 'empty',
      child_process: 'empty'
    },
    // 不启用 Webpack 内置的优化策略
    performance: false
  };
};

webpackDevServer.config.js

'use strict';

const fs = require('fs');
// 可以直接通过报错打开本地的编辑器并跳转到对应的位置，在 WebpackDevServer 的 before 钩子上
const errorOverlayMiddleware = require('react-dev-utils/errorOverlayMiddleware');
const evalSourceMapMiddleware = require('react-dev-utils/evalSourceMapMiddleware');
// 避免在开发环境中使用生产版本到的 service-worker.js，返回一个使用重置的 service worker 配置
const noopServiceWorkerMiddleware = require('react-dev-utils/noopServiceWorkerMiddleware');
// 返回一个忽略 node_modules 的正则表达式
const ignoredFiles = require('react-dev-utils/ignoredFiles');
const redirectServedPath = require('react-dev-utils/redirectServedPathMiddleware');
const paths = require('./paths');
const getHttpsConfig = require('./getHttpsConfig');
const host = process.env.HOST || '0.0.0.0';
const sockHost = process.env.WDS_SOCKET_HOST;
const sockPath = process.env.WDS_SOCKET_PATH;
const sockPort = process.env.WDS_SOCKET_PORT;

module.exports = function (proxy, allowedHost) {
  return {
    // 设置为 true 时，此选项绕过主机检查；不检查主机的应用程序容易受到 DNS 重新连接攻击
    disableHostCheck:
      !proxy || process.env.DANGEROUSLY_DISABLE_HOST_CHECK === 'true',
    // 启用 gzip 压缩
    compress: true,
    // CRA 的意思是，这玩意没什么卵用
    clientLogLevel: 'none',
    // 告诉服务器从哪个目录中提供内容
    contentBase: paths.appPublic,
    contentBasePublicPath: paths.publicUrlOrPath,
    // 开启此选项后，在文件修改之后，会触发一次完整的页面重载
    watchContentBase: true,
    // 当 CSS 文件修改时 CRA 会 HMR，JS 文件修改时会刷新页面
    hot: true,
    // 使用 Websocket 替代 sockjs-node
    transportMode: 'ws',
    injectClient: false,
    sockHost,
    sockPath,
    sockPort,
    // 此路径下的打包文件可在浏览器中访问
    publicPath: paths.publicUrlOrPath.slice(0, -1),
    // 除了初始启动信息之外的任何内容都不会被打印到控制台；Webpack 的错误或警告在控制台不可见
    quiet: true,
    // 对于某些系统，监听大量文件系统会导致大量的 CPU 或内存占用，这个选项可以排除一些巨大的文件夹
    watchOptions: {
      ignored: ignoredFiles(paths.appSrc)
    },
    https: getHttpsConfig(),
    host,
    // 当出现编译器错误或警告时，在浏览器中显示全屏覆盖层；默认禁用
    overlay: false,
    historyApiFallback: {
      disableDotRule: true,
      index: paths.publicUrlOrPath
    },
    public: allowedHost,
    // proxy 将在 WDS 的 before、after 钩子之间运行
    proxy,
    // 在服务内部的所有其他中间件之前，提供执行自定义中间件的功能，这可以用来配置自定义处理程序
    before(app, server) {
      app.use(evalSourceMapMiddleware(server));
      app.use(errorOverlayMiddleware());

      if (fs.existsSync(paths.proxySetup)) {
        require(paths.proxySetup)(app);
      }
    },
    // 在服务内部的所有其他中间件之后，提供执行自定义中间件的功能
    after(app) {
      app.use(redirectServedPath(paths.publicUrlOrPath));
      app.use(noopServiceWorkerMiddleware(paths.publicUrlOrPath));
    }
  };
};
