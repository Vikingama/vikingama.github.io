如何在组件卸载时取消请求？

2020.05.10

最近接触到了一些复杂的 Vue 项目，发现在组件销毁时，组件里的请求也会取消。

说实话，自己之前没有注意过这个问题；一来是因为单打独斗惯了，没有工程化的思想；二来自己的 HTTP 知识浅薄，没有遭遇极端情况。

API

XMLHttpRequest、fetch 和 axios 都提供了取消请求的方法：

// XMLHttpRequest
var xhr = new XMLHttpRequest();
xhr.open("GET", "https://developer.mozilla.org/", true);
xhr.send();
xhr.abort();
// fetch
var controller = new AbortController();
var signal = controller.signal;
setTimeout(() => {
    controller.abort();
}, 100);
fetch("https://developer.mozilla.org/", {
    method: "GET",
    signal,
})
    .then((res) => res.json())
    .then((json) => {
        console.log(json);
    })
    .catch((err) => {
        console.error(err);
    });
// axios
var CancelToken = axios.CancelToken;
var source = CancelToken.source();
axios
    .post(
        "https://developer.mozilla.org/",
        {
            name: "mozilla",
        },
        {
            cancelToken: source.token,
        }
    )
    .then((res) => {
        console.info(res);
    })
    .catch((info) => {
        if (axios.isCancel(info)) {
            console.log(info);
        } else {
            console.error(info);
        }
    });
source.cancel("...");

Vue

这篇文章的最后的例子很有启发性；这个组件返回了一个作用域插槽；通过组件 props 接收请求地址、参数和方法，将请求状态和返回数据注入插槽：

Vue.component("fetch-data", {
    props: {
        url: {
            type: String,
            required: true,
        },
        method: {
            type: String,
            default: "GET",
        },
        params: {
            type: Object,
            default: {},
        },
        body: {
            type: Object,
            default: {},
        },
    },
    data() {
        return {
            loaded: false,
            response: null,
            error: null,
        };
    },
    created() {
        var CancelToken = axios.CancelToken;
        var source = CancelToken.source();
        axios({
            url: this.url,
            method: this.method.toUpperCase(),
            params: this.params,
            data: this.body,
            cancelToken: source.token,
        })
            .then((response) => {
                this.response = response;
                this.loaded = true;
            })
            .catch((error) => {
                this.error = error;
                this.loaded = true;
            });
        // 在组件销毁之前如果请求没有完成，则取消请求
        this.$once("hook:beforeDestroy", () => {
            source.cancel("...");
        });
    },
    render() {
        if (!this.loaded) {
            return this.$scopedSlots.default({
                loading: true,
                response: null,
                error: null,
            });
        }
        return this.$scopedSlots.default({
            loading: false,
            response: this.response.data,
            error: this.error,
        });
    },
});

React

React 可以通过高阶组件或者 Hook 实现：

// HOC 伪代码
const withFetchData = (Comp) =>
    class WithFetchData extends React.Component {
        ct = axios.CancelToken.source();
        state = {
            isLoading: true,
            res: null,
            err: null,
        };
        loadData = () => {
            const cancelToken = this.ct.token;
            // 请求方式、参数、地址可以通过 Props 传入
            const { apiConf } = this.props;
            axios({
                ...apiConf,
                cancelToken,
            })
                .then((response) => {
                    this.setState({
                        isLoading: false,
                        res: response,
                    });
                })
                .catch((error) => {
                    this.setState({
                        isLoading: false,
                        err: error,
                    });
                });
        };
        shouldComponentUpdate(nextProps, nextState) {
            // 如果请求参数变化就重新发起请求
            return !isEqual(this.props, nextProps);
        }
        componentDidMount() {
            this.loadData();
        }
        componentDidUpdate() {
            // 重新请求之前取消上次请求
            this.ct.cancel("...");
            this.setState({
                isLoading: true,
                res: null,
                err: null,
            });
            this.loadData();
        }
        componentWillUnmount() {
            this.ct.cancel("...");
        }
        render() {
            const { isLoading, res, err } = this.state;
            if (isLoading) {
                return <Loading />;
            }
            if (res) {
                return <Comp {...this.props} data={res} />;
            }
            if (err) {
                return <Error />;
            }
        }
    };
// Hook 伪代码
const useFetch = (apiConf) => {
    const [isLoading, setIsLoading] = useState(true);
    const [res, setRes] = useState(null);
    const [err, setErr] = useState(null);
    const ct = useRef();
    useEffect(() => {
        ct.current = axios.CancelToken.source();
    }, []);
    useEffect(() => {
        setIsLoading(true);
        axios({
            ...apiConf,
            cancelToken: ct.current.token,
        })
            .then((response) => {
                setRes(response);
            })
            .catch((error) => {
                setErr(error);
            })
            .finally(() => {
                setIsLoading(false);
            });
        return () => {
            ct.current.cancel("...");
        };
    }, [apiConfig]);
    return {
        isLoading,
        res,
        err,
    };
};

当然，取消请求只适用于获取数据的情况；当用户进行增删改等关键操作时，请求就不应该被取消。
